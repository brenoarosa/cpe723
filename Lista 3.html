def calculate_D(X, tresh):
    X1 = X[X <= tresh]
    X2 = X[X > tresh]
    
    y1 = X1.mean()
    y2 = X2.mean()
    
    d1 = (X1 - y1)**2
    d2 = (X2 - y2)**2
    
    D = np.hstack((d1, d2))
    
    return D.mean()

1.g)¶

Observamos que ao utilizarmos temperaturas mais altas o processo de soft clustering de cada elemento x obtêm probabilidades de clusters mais próximas de equiprovaveis. Resultante disso, atenta-se que os centroídes dos clusters ficam mais próximos entre si.

A medida que abaixamos a temperatura, a clusterização vai se aproximando de um processo de hard clustering, no qual cada elemento x tem probabilidade quase absoluta de pertencer a um dado cluster y. Neste caso, quanto menor a temperatura mais os centroides dos clusters se aproximam dos centroides obtidos por hard clustering.

Ressalta-se que o custo obtido se aproxima do custo mínimo a medida que a temperatura abaixa.

def J(X, Y):
    """
    Calcula norma l2 média entre ponto e centro do cluster
    """
    d = np.zeros((Y.shape[0], X.shape[0])) # distance between point and clusters
    for i, y in enumerate(Y):
        d[i, :] = np.sqrt(((X - y)**2).sum(axis=1))
        
    return d


def clustering_deterministic_anneling(X, n_clusters, J, N, K, t0, alpha=0.95):
    """
    X: posição dos pontos
    n_clusters: número de clusters
    J: função custo
    N: número de iterações por temperatura
    K: número de temperaturas testadas
    t0: temperatura inicial
    alpha: decaimento da temperatura
    """
    
    sample_size = X.shape[0]
    
    S = np.zeros((K*N+1, n_clusters, 2), dtype=float) # matriz de historico dos clusters
    D = np.zeros((K*N), dtype=float) # matriz de historico dos custos
    Y = np.random.normal(0, 1, size=(n_clusters, 2)) # estado inicial
    S[0, :, :] = Y.copy()
    
    idx = 0
    for k in range(K):
        t = alpha**k * t0
        for _ in range(N):
            cost = J(X, Y)

            Z = np.exp(-cost / t).sum(axis=0)

            py_given_x = np.exp(-cost / t) / Z

            D[idx] = (py_given_x * cost).sum() / len(X)
            
            py = py_given_x.sum(axis=1)
            
            for i, y in enumerate(Y):
                Y[i, :] = (py_given_x[i].reshape(-1, 1) * X).sum(axis=0) / py[i]
                
            S[idx+1, :, :] = Y.copy()
            idx += 1

    # Retorna historico de estados
    return S, D
