def simulated_anneling(J, N, T, s0):
    """
    J: vetor de custo associado a cada estado
    N: nmero de iteraes de metropolis
    T: vetor de temperaturas utilizadas
    s0: estado inicial
    """
    
    X = np.zeros((N*len(T) + 1), dtype=int) # vetor com historico dos estados
    X[0] = s0 - 1 # transforma de estado para indice
    
    i = 0
    for t in T:
        for _ in range(N):
            x_candidate = np.random.randint(low=0, high=5)
            delta_J = J[x_candidate] - J[X[i]]
            q = np.exp(-delta_J/t)
            r = np.random.uniform(low=0., high=1.)

            if r < q:
                X[i+1] = x_candidate
            else:
                X[i+1] = X[i]
                
            i = i + 1

    S = X + 1 # transforma de indice para estados
    return S

def cluster_J(X, y, n_clusters):
    """
    Calcula norma l2 mdia entre ponto e centro do cluster
    """
    
    d = np.zeros(X.shape) # distance between point and clusters
    for i in range(n_clusters):
        c = X[y == i].mean(axis=0) # cluster center
        d[y == i] = X[y == i] - c
    
    return np.sqrt((d**2).sum(axis=1)).mean()


def clustering_simulated_anneling(X, n_clusters, J, N, K, t0):
    """
    X: posio dos pontos
    n_clusters: mero de clusters
    J: funo custo
    N: nmero de iteraes do algoritmo de metropolis
    K: nmero de iteraes do simulated anneling
    t0: temperatura inicial
    """
    
    sample_size = X.shape[0]
    
    Y = np.zeros((K*N+1, sample_size), dtype=int) # matriz de historico dos estados
    Y[0] = np.random.randint(n_clusters, size=sample_size) # estado inicial
    
    i = 0
    for k in range(1, K+1):
        t = t0 / np.log2(1+k)
        
        for _ in range(N):
            
            # define perturbao
            cand_y = Y[i, :].copy()
            cand_idx = np.random.randint(low=0, high=sample_size)
            cand_cluster = np.random.randint(n_clusters)
            cand_y[cand_idx] = cand_cluster
            
            delta_J = J(X, cand_y, n_clusters) - J(X, Y[i, :], n_clusters)
            q = np.exp(-delta_J/t)
            r = np.random.uniform(low=0., high=1.)

            if r < q:
                Y[i+1] = cand_y
            else:
                Y[i+1] = Y[i]
                
            i = i + 1

    # Retorna historico de estados
    return Y
